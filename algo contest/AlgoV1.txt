INPUT: 3 entier n1,n2,n3
OUTPUT:Une chaine de symbole représentant un morceau du sapin

PROCEDURE print_tree(n1,n2,n3)

	VAR 
	

	BEGIN
		FOR size <-- 1 TO n1 DO
			print ((size * "*").center(23));
			size += 2;
		FOR size <-- 3 TO n2 DO
			print ((size * "*").center(23));
			size += 4;
		FOR size <-- 5 TO n3 DO
			print ((size * "*").center(23));
			size += 6;
	END

print_tree(8,16,24)
